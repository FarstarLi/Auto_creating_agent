# ============================================================
# 安全限制配置
# - [commands] 下的命令禁止执行
# - [files] 下的路径/文件名禁止读取
# - 支持 * 通配符（glob 匹配）
# - 不含 * 的项以独立命令词匹配（不会被 URL 参数等子串误伤）
# - 以 # 开头的行为注释
# ============================================================

[commands]

rm -rf /*
del /f /s C:\
del /f /s C:\Windows

shutdown *
sudo rm *
mkfs*
dd if=*
> /dev/sda
:(){ :|:& };:
:fork
while true

.env
*.key

*.p12
